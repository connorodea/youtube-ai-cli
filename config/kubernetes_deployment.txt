apiVersion: apps/v1
kind: Deployment
metadata:
  name: youtube-ai-cli
  namespace: youtube-ai
  labels:
    app: youtube-ai-cli
    version: v1
spec:
  replicas: 3
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxUnavailable: 1
      maxSurge: 1
  selector:
    matchLabels:
      app: youtube-ai-cli
  template:
    metadata:
      labels:
        app: youtube-ai-cli
        version: v1
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8080"
        prometheus.io/path: "/metrics"
    spec:
      serviceAccountName: youtube-ai-cli
      securityContext:
        runAsNonRoot: true
        runAsUser: 1000
        fsGroup: 1000
      containers:
      - name: youtube-ai-cli
        image: yourusername/youtube-ai-cli:latest
        imagePullPolicy: Always
        ports:
        - containerPort: 8080
          name: metrics
          protocol: TCP
        - containerPort: 8081
          name: health
          protocol: TCP
        env:
        - name: OPENAI_API_KEY
          valueFrom:
            secretKeyRef:
              name: youtube-ai-secrets
              key: openai-api-key
        - name: ANTHROPIC_API_KEY
          valueFrom:
            secretKeyRef:
              name: youtube-ai-secrets
              key: anthropic-api-key
        - name: YOUTUBE_API_KEY
          valueFrom:
            secretKeyRef:
              name: youtube-ai-secrets
              key: youtube-api-key
        - name: ELEVENLABS_API_KEY
          valueFrom:
            secretKeyRef:
              name: youtube-ai-secrets
              key: elevenlabs-api-key
              optional: true
        - name: OUTPUT_DIR
          value: "/app/output"
        - name: DEBUG
          value: "false"
        - name: LOG_LEVEL
          value: "INFO"
        - name: ENABLE_METRICS
          value: "true"
        - name: METRICS_PORT
          value: "8080"
        - name: ENABLE_HEALTH_CHECK
          value: "true"
        - name: HEALTH_CHECK_PORT
          value: "8081"
        - name: DATABASE_URL
          valueFrom:
            secretKeyRef:
              name: youtube-ai-secrets
              key: database-url
        volumeMounts:
        - name: output-storage
          mountPath: /app/output
        - name: config-storage
          mountPath: /home/app/.youtube-ai
        - name: cache-storage
          mountPath: /app/cache
        - name: credentials
          mountPath: /app/credentials
          readOnly: true
        resources:
          requests:
            memory: "512Mi"
            cpu: "250m"
          limits:
            memory: "2Gi"
            cpu: "1"
        livenessProbe:
          httpGet:
            path: /health
            port: 8081
          initialDelaySeconds: 30
          periodSeconds: 10
          timeoutSeconds: 5
          failureThreshold: 3
        readinessProbe:
          httpGet:
            path: /ready
            port: 8081
          initialDelaySeconds: 5
          periodSeconds: 5
          timeoutSeconds: 3
          failureThreshold: 3
        startupProbe:
          httpGet:
            path: /health
            port: 8081
          initialDelaySeconds: 10
          periodSeconds: 5
          timeoutSeconds: 3
          failureThreshold: 30
      volumes:
      - name: output-storage
        persistentVolumeClaim:
          claimName: youtube-ai-output-pvc
      - name: config-storage
        persistentVolumeClaim:
          claimName: youtube-ai-config-pvc
      - name: cache-storage
        emptyDir:
          sizeLimit: 1Gi
      - name: credentials
        secret:
          secretName: youtube-ai-credentials
          defaultMode: 0400
      tolerations:
      - key: "app"
        operator: "Equal"
        value: "youtube-ai"
        effect: "NoSchedule"
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
          - weight: 100
            podAffinityTerm:
              labelSelector:
                matchExpressions:
                - key: app
                  operator: In
                  values:
                  - youtube-ai-cli
              topologyKey: kubernetes.io/hostname
